1、nginx 转发 sftp
    添加以下配置：
        stream {    
            upstream sftp{
                hash $remote_addr consistent; 
                server 192.168.137.104:22 max_fails=3 fail_timeout=30s; 
            }
            server {
                listen 9222;
                proxy_connect_timeout 1s; 
                proxy_timeout 30s; 
                proxy_pass sftp;
            }
        }
